# 数据库导入导出

创建数据库

CREATE DATABASE `bi_ott` DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

连接远程数据库

bin/mysql -uroot -S mysql.sock -h132.225.1.188 -p

导出数据库

bin/mysqldump -uroot -S mysql.sock -h127.0.0.1 -p  数据库名称 > /data/数据库名称.sql

导出所有表结构

bin/mysqldump -uroot -S mysql.sock -h127.0.0.1 -p --opt -d 数据库名称 > /data/数据库名称.sql

导出单个表

bin/mysqldump -uroot -S mysql.sock -h127.0.0.1 -p 数据库名称 表名 > /data/表名.sql

导入数据库

# 首先切换到对应的数据库
source /data/mydb.sql
