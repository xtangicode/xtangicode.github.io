# 1. 创建目录

[root@txSugar minio]# cd /data
[root@txSugar minio]# mkdir minio

# 2. 下载MinIO

[root@txSugar minio]# wget https://dl.minio.io/server/minio/release/linux-amd64/minio
[root@txSugar minio]# chmod 777 minio

# 3. 创建 minio.log

[root@txSugar minio]# cd minio/
[root@txSugar minio]# touch minio.log

# 4. 启动服务

指定存放静态文件位置，可以使用--console-address :50000 --address :9000 指定端口启动

[root@txSugar minio]# ./minio server /data/minio/data

# 5. 修改默认登录名、密码

[root@txSugar ~]# vim /etc/profile

写入如下配置

export MINIO_ROOT_USER=fileadmin
export MINIO_ROOT_PASSWORD=fileadmin

[root@txSugar ~]# source /etc/profile

# 6. 重启服务
