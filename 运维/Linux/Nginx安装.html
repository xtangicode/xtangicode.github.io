# 下载Nginx

[root@txSugar]# cd /opt/ 
[root@txSugar opt]# wget https://nginx.org/download/nginx-1.20.2.tar.gz

# 解压安装

[root@txSugar opt]# tar -zxvf nginx-1.20.2.tar.gz
[root@txSugar opt]# cd nginx-1.20.2 
[root@txSugar opt]# ./configure --prefix=/usr/local/nginx --with-http_stub_status_module --with-http_ssl_module
[root@txSugar opt]# make 
[root@txSugar opt]# make install
[root@txSugar opt]# ln -s /usr/local/nginx/sbin/nginx /usr/local/bin/nginx

# 启动服务

[root@txSugar opt]# cd /usr/local/nginx/sbin/
[root@txSugar sbin]# ./nginx

# 测试

浏览器输入localhost默认端口80，无法访问检查nginx是否启动成功或者端口是否开放

# 添加nginx-http-flv-module模块

举例比如安装nginx-http-flv-module，下载地址：https://github.com/winshining/nginx-http-flv-module/archive/refs/tags/v1.2.9.tar.gz 未安装nginx

  1. 则执行./configure --add-module=../需要添加的模块目录，例如：./configure --add-module=../nginx-http-flv-module-master
  2. 执行make编译，最后make install安装

已安装nginx

  1. 查看nginx安装的版本及依赖信息nginx -V
[root@txSugar bin]# nginx -V
nginx version: nginx/1.22.1
built by gcc 4.8.5 20150623 (Red Hat 4.8.5-44) (GCC) 
built with OpenSSL 1.0.2k-fips  26 Jan 2017
TLS SNI support enabled
configure arguments: --prefix=/usr/local/nginx --with-http_stub_status_module --with-http_ssl_module
[root@txSugar bin]#

  1. 如果本地还保留之前的安装包（源码包），则直接进入执行命令，否则下载对应的nginx。如上nginx版本1.22.1。

  2. 进入源码包，执行命令

[root@txSugar bin]# cd nginx-1.22.1
[root@txSugar nginx-1.22.1]# --prefix=/usr/local/nginx --with-http_stub_status_module --with-http_ssl_module --add-module=../nginx-http-flv-module-master
  1. 编译make，编译后千万不可以执行make install，否则会将已安装的文件覆盖。
  2. 替换nginx包
[root@txSugar nginx-1.22.1]# which nginx
/usr/sbin/nginx
[root@txSugar nginx-1.22.1]# cp  /usr/sbin/nginx /usr/sbin/nginx.bak
[root@txSugar nginx-1.22.1]# cp -f objs/nginx /usr/sbin/nginx
cp: overwrite ‘/usr/sbin/nginx’? y
  1. 查询依赖，检查是否安装完成
[root@txSugar nginx-1.22.1]# nginx -V
nginx version: nginx/1.22.1
built by gcc 4.8.5 20150623 (Red Hat 4.8.5-44) (GCC)
