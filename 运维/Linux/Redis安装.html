# 准备环境

官网下载redis-5.0.8.tar.gz

# 解压安装，配置端口

  1. 将redis-5.0.8.tar.gz 放到/data下

  2. tar -zxvf redis-5.0.8.tar.gz 解压到当前文件夹，

  3. cd redis-5.0.8/

  4. 没有安装gcc需要先安装gcc

yum install gcc
make PREFIX=/data/redis-5.0.8 install
  1. 修改redis.conf，先备份cp redis.conf redis.conf.back
bind 0.0.0.0  #允许任意ip访问
daemonize yes #守护进程，可以再后台运行redis
requirepass 11122 #redis连接密码。
port 16379

  1. bin/redis-server ./redis.conf 启动

  2. bin/redis-cli -p 16379

  3. auth 11122 密码认证

  4. keys *

# 配置成服务

配置自启动 vi /etc/init.d/redis16379d

#!/bin/bash
#chkconfig:345 61 61
#description:redis16379d
/data/redis-5.0.8/bin/redis-server /data/redis-5.0.8/redis.conf

chkconfig --add redis16379d
chkconfig --list
chmod 777 redis16379d  赋予权限
service redis16379d start  测试
ps -ef|grep redis
