# 确定本机系统内核

本机系统是CentOS 7

[root@localhost docker]# cat /etc/redhat-release 
CentOS Linux release 7.4.1708 (Core)

或者

[root@localhost docker]# uname -a
Linux localhost.localdomain 3.10.0-693.el7.x86_64 #1 SMP Tue Aug 22 21:09:27 UTC 2017 x86_64 x86_64 x86_64 GNU/Linux

# 卸载旧版本

较旧的Docker版本称为docker或docker-engine。如果已安装这些程序，请卸载它们以及相关的依赖项。/var/lib/docker/下的镜像、容器等文件也需要可以自行删除。

 yum remove docker \
             docker-client \
             docker-client-latest \
             docker-common \
             docker-latest \
             docker-latest-logrotate \
             docker-logrotate \
             docker-engine

# 安装依赖库

yum install -y yum-utils

# 设置稳定版本的镜像仓库

yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

# 更新yum软件包索引

yum makecache fast

# 安装Docker Engine

# 配置Hub国内镜像

  1. 如果拉取镜像失败，需要配置国内的镜像
vi /etc/docker/daemon.json
  1. 配置网易163的hub镜像，也可以配置国内其他镜像，比如阿里云镜像
{"registry-mirrors":["http://hub-mirror.c.163.com"]}
  1. 重启服务
systemctl daemon-reload
systemctl restart docker
  1. 再次执行
docker run hello-world

# 配置阿里云镜像

  1. 打开注册地址 (opens new window)，使用淘宝或者支付宝账号登录
  2. 选择控制台
  3. 搜索容器镜像服务
  4. 复制镜像地址
  5. 修改daemon.json
vi /etc/docker/daemon.json

{"registry-mirrors":["https://xxxx.mirror.aliyuncs.com"]}
  1. 重启服务
systemctl daemon-reload
systemctl restart docker
  1. 测试拉取镜像
docker pull tomcat

# run命令干了哪些事情

  1. 检查本地是否有hello-world镜像，有则以改镜像为模板创建容器实例运行。

  2. 没有则从Docker Hub拉取（默认从国外的Hub拉取，配置了国内镜像则拉取国内Hub镜像），相当于执行命令docker pull hello-world。

  3. 将镜像为模板创建容器实例运行，相当于执行命令docker container create。

# 卸载

systemctl stop docker
yum remove docker-ce docker-ce-cli containerd.io
rm -rf /var/lib/docker
rm -rf /var/lib/containerd

# 底层原理

docker使用Go语言编写，直接使用物理机的硬件资源，并且直接使用宿主机的内核。所以各方面性能都比虚拟机快。虚拟机需要虚拟硬件，在宿主机上虚拟一套系统。
