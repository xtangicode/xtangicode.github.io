# SFTP

文件上传、下载、删除

# 引入相关依赖

</dependency>

# 代码实现

public static void main(String[] args) {
        @Cleanup Sftp sftp = new Sftp("127.0.0.1", 22, "sftpuser", "sftppwd");
        String sftpPath = "/data/sugar/test";
        List<String> sftpFiles = sftp.ls(sftpPath, (sftpFile) ->
            // 已test开头的文件或者目录
            sftpFile.getFilename().startsWith("test")
        );
        File tempPath = new File("/opt/download/tempfile/" + DateUtil.format(new Date(), DatePattern.PURE_DATETIME_PATTERN) + "/");
        tempPath.mkdirs();
        for (String sftpFile : sftpFiles) {
            //下载文件
            sftp.download(sftpPath + sftpFile,  FileUtil.file(tempPath.getPath()));
            log.info("成功下载文件名：{}", sftpFile);
        }
        File[] tempFiles = tempPath.listFiles();
        for (File tempFile : tempFiles) {
            FileUtil.readLines(tempFile, Charset.forName("UTF-8"), (LineHandler) (String line) ->
                    {
                        if (StrUtil.isNotBlank(line)) {
                            System.out.println(line);                            
                        }
                    }
            );
        }
    }

# FTP

文件上传下载删除 使用的是hutool中自带的ftp工具实现，所以需要提前引入hutool的依赖。

# 引入相关依赖

</dependency>
    <groupId>com.jcraft</groupId>
    <artifactId>jsch</artifactId>
    <version>0.1.55</version>
</dependency>
<dependency>
    <groupId>cn.hutool</groupId>
    <artifactId>hutool-all</artifactId>
    <version>5.7.20</version>
</dependency>
<dependency>
    <groupId>org.projectlombok</groupId>
    <artifactId>lombok</artifactId>
    <version>1.18.4</version>
    <scope>provided</scope>
</dependency>

# 代码实现

public static void main(String[] args) {
    @Cleanup
    Ftp ftp = new Ftp("127.0.0.1", 21, "ftpuser", "ftppwd");
    ftp.setMode(FtpMode.Passive);
    List<FTPFile> files = ftp.lsFiles("/ftpdir/", (Filter<FTPFile>) (file) -> {
        return file.getName().startsWith("test");
    });
    for (FTPFile file : files) {
        ftp.download("/ftpdir/" +file.getName(),  FileUtil.file("/home/dev/tx/"));
        log.info("文件：{}下载成功", file.getName());
    }
}
