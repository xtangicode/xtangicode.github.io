在进行接口时，我们经常需要获取当前请求的用户信息，直接传递用户id又不是很安全。可以采用如下方案

# 登录时返回有效期token

token可以使用jwt生成（注意：token中切勿存敏感信息），或者自定义的随机字符串，使用key-value的格式将用户信息保存在本地缓存或者redis中。

# 拦截token

请求接口通过拦截器/过滤器验证是否包含token，也可以将部分接口加入不需要验证token的白名单。

# 验证有效期

验证token的有效期，可以通过token本身的有效时间，或者通过本地缓存的有效时间进行验证

# 验证用户信息及权限

验证缓存中是否有此token对应的用户，是否有此接口的访问/操作权限。有则通过ThreadLocal进行保存放行，否则报错返回给前台。

# 请求过程中获取上下文用户

直接获取ThreadLocal中保存的用户信息，原因：ThreadLocal是以每个线程作为key，数据作为value进行存放，所以线程与线程之间是隔离的，因此请求结束后必须得情况当先线程的ThreadLocal，否则请求过多容易OOM。

package com.sugar.framework.interceptor;
import cn.hutool.core.bean.BeanUtil;
import com.alibaba.fastjson.JSON;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.unicom.common.exception.BusinessException;
import com.unicom.common.utils.*;
import com.unicom.framework.ctx.Ctx;
import com.unicom.framework.ctx.CtxUser;
import com.unicom.framework.web.domain.AjaxResult;
import io.jsonwebtoken.Claims;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.lang.Nullable;
import org.springframework.stereotype.Component;
import org.springframework.web.method.HandlerMethod;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

/**
 * @author: sugar
 */
@Slf4j
@Component
public class SSOInterceptor implements HandlerInterceptor{

    @Autowired
    ObjectMapper objectMapper;
    @Autowired
    private JwtTokenUtil jwtTokenUtil;

    @Value("${api.sso.exclude-urls}")
    private String excludeUrls;
    @Value("${api.sso.flag}")
    private boolean flag;

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception
    {
        boolean access = true;
        String errorMsg = "";
        if (handler instanceof HandlerMethod)
        {
//            HandlerMethod handlerMethod = (HandlerMethod) handler;
//            Method method = handlerMethod.getMethod();
            String token = request.getHeader("Access-token");
            String servletPath = request.getServletPath();
            List<String> excludeUrlArr = new ArrayList<>();
            if (ToolUtil.isNotEmpty(excludeUrls)) {
                excludeUrlArr = Arrays.asList(Convert.toStrArray(",", excludeUrls));
            }
//            log.info("token：" + token);
            if (flag) {
               if (token != null) {
                   try {
                       if (jwtTokenUtil.isTokenExpired(token)) {
                           throw new BusinessException("token已过期");
                       }
                       jwtTokenUtil.parseToken(token);
                       CtxUser userInfo = BeanUtil.toBean(CacheUtil.get(token), CtxUser.class);
                       Ctx.set(userInfo);
                   } catch (Exception e) {
                       errorMsg = "Access-token解析异常!";
                       access = false;
                   }
               } else {
                   if (!excludeUrlArr.contains(servletPath)) {
                       errorMsg = "请求head中缺少Access-token!";
                       access = false;
                   }
               }
            }
        }
        if (!access) {
            ServletUtils.renderString(response, JSON.toJSONString(AjaxResult.error(errorMsg)));
        }
        return access;

    }

    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response, Object handler, @Nullable Exception ex) throws Exception {
        // 移除UserInfo防止ThreadLocal OOM
        Ctx.clean();
    }

}

package com.sugar.framework.config;

import com.unicom.framework.interceptor.SSOInterceptor;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.*;

/**
 * 通用配置
 * 
 * @author sugar
 */
@Configuration
public class ResourcesConfig implements WebMvcConfigurer
{

    @Autowired
    private SSOInterceptor ssoInterceptor;

    static final String ORIGINS[] = new String[] { "GET", "POST", "PUT", "DELETE" };


    @Override
    public void addCorsMappings(CorsRegistry registry) {
        registry.addMapping("/**").allowedOrigins("*").allowCredentials(true).allowedMethods(ORIGINS).maxAge(3600);
    }


    @Override
    public void addResourceHandlers(ResourceHandlerRegistry registry)
    {
        /** 文件上传路径 */
        registry.addResourceHandler("/profile/**").addResourceLocations("file:" + UnicomConfig.getProfile());
        /** swagger配置 */
        registry.addResourceHandler("swagger-ui.html").addResourceLocations("classpath:/META-INF/resources/");
        registry.addResourceHandler("/webjars/**").addResourceLocations("classpath:/META-INF/resources/webjars/");
    }

    /**
     * 自定义拦截规则
     */
    @Override
    public void addInterceptors(InterceptorRegistry registry)
    {
        registry.addInterceptor(ssoInterceptor).addPathPatterns("/**");
    }
}

package com.sugar.framework.ctx;

/**
 * @author: sugar
 */
public class Ctx {

    private static ThreadLocal<CtxUser> threadLocal = new ThreadLocal<>();

    public static void set(CtxUser ctxUser) {
        threadLocal.set(ctxUser);
    }

    public static void clean() {
        if (threadLocal != null && threadLocal.get() != null) {
            threadLocal.remove();
        }
    }

    public static CtxUser getUser() {
        return threadLocal.get();
    }
}
